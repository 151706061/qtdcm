181 void QtDcm::serieItemSelected ( QTreeWidgetItem* current, QTreeWidgetItem* previous )
182 {

184  {
185  QtDcmManager::instance()->clearListOfImages();
186 
187  if ( d->mode == QtDcm::CD )
188  QtDcmManager::instance()->findImagesDicomdir ( current->text ( 3 ) );
189  else
190  QtDcmManager::instance()->findImagesScu ( current->text ( 3 ) );
191

193  QString institution = current->data ( 5, 0 ).toString();

195 
196  QtDcmManager::instance()->updateSerieInfo ( QString::number ( elementCount ), institution, opName );
197 // QtDcmManager::instance()->clearPreview();
198  QtDcmManager::instance()->getPreviewFromSelectedSerie ( current->text ( 3 ), elementCount / 2 );
199  }

210 void QtDcm::openDicomdir()
211 {
212  this->clearDisplay();
213  d->mode = QtDcm::CD;
214  // Open a QFileDialog for choosing a Dicomdir
215  QFileDialog dialog(this);
216  dialog.setWindowTitle ( tr ( "Open dicomdir" ) );
217  dialog.setNameFilters(QStringList() << "Dicomdir files (dicomdir* DICOMDIR*)");
218 
219  // Trying to open directly on one of the available drives
220  if (!QDir::drives().isEmpty()) {
221  dialog.setDirectory ( QDir::drives().first().absoluteDir() );
222  }
223 
224  QString fileName;
225  if ( dialog.exec() ) {
226  fileName = dialog.selectedFiles() [0];
227  }
228 
229  if ( !fileName.isEmpty() ) { // A file has been chosen{
230  if (QString::compare(fileName, "dicomdir", Qt::CaseInsensitive)) {
231  QtDcmManager::instance()->setDicomdir ( fileName );
232  this->loadPatientsFromDicomdir();
233  }
234  }

248 void QtDcm::updateModality ( int index )

254  break;
255 
256  case 1://MR

264  case 3://PET
265  QtDcmManager::instance()->setModality ( "PET" );
266  break;
267  }
268 
269  treeWidgetSeries->clear();
270 
271  if ( treeWidgetPatients->currentItem() && treeWidgetStudies->currentItem() ) {
272  if ( d->mode == QtDcm::PACS ) {
273  QtDcmManager::instance()->findSeriesScu ( treeWidgetPatients->currentItem()->text ( 0 ), treeWidgetStudies->currentItem()->data ( 2, 0 ).toString() );
274  }
275  else {
276  QtDcmManager::instance()->findSeriesDicomdir ( treeWidgetPatients->currentItem()->text ( 0 ), treeWidgetStudies->currentItem()->data ( 2, 0 ).toString() );
277  }
